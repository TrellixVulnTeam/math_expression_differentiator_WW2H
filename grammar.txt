<expression> ::= <term>
               | <expression> + <term>
               | <expression> - <term>

<term> ::= <factor>
         | <term> * <factor>
         | <term> / <factor>

<factor> ::= <func>
           | - <factor>
           | + <factor>
           | <power>

<power> ::= <func> '^' <factor>

<func> ::= <primary>
         | <'sin('primary')'>
         | <'cos('primary')'>
         | <'tan('primary')'>
         | <'exp('primary')'>
         | <'ln('primary')'>
         | <'log('primary')'>
         | <'func('primary')'>
         | <'sqrt('primary')'>

<primary> ::= <IDENTIFIER>
            | <INTEGER>
            | <FLOATING_POINT_LITERAL>
            | <'(' expression ')'>

PROGRAM ::= OPTFUNCS OPTDIFFARG OPTEXPRS

OPTFUNCS ::= EMPTY | FUNCS

OPTDIFFARG ::= EMPTY | DIFFARG

OPTEXPRS ::= EMPTY | EXPRS

FUNCS ::= FUNC FUNCS | FUNC

FUNC ::= def id '(' id ')' '{' EXPR '}' 

DIFFARG ::= '{' diffby id '}'

EXPRS ::= '{' EXPR '}' EXPRS | EXPR

EXPR ::=   EXPR + EXPR
         | EXPR - EXPR
         
         | EXPR * EXPR
         | EXPR / EXPR
         
         | EXPR ^ EXPR

         | - EXPR
         | + EXPR

         | sin  EXPR
         | cos  EXPR
         | tan  EXPR
         | exp  EXPR
         | ln   EXPR
         | log  EXPR
         | sqrt EXPR
         | func EXPR
         
         | '(' EXPR ')'
         
         | number
         | id
